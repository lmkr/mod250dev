19 package org.foo.shape.circle;
20 
21 import java.awt.*;
22 import java.awt.geom.Ellipse2D;
23 import org.foo.shape.SimpleShape;
24 
25 public class Circle implements SimpleShape {
26 
27   /**
28    * Implements the <tt>SimpleShape.draw()</tt> method for painting the shape.
29    * 
30    * @param g2 The graphics object used for painting.
31    * @param p The position to paint the triangle.
32    **/
33   public void draw(Graphics2D g2, Point p) {
34     int x = p.x - 25;
35     int y = p.y - 25;
36     GradientPaint gradient = new GradientPaint(x, y, Color.RED, x + 50, y, Color.WHITE);
37     g2.setPaint(gradient);
38     g2.fill(new Ellipse2D.Double(x, y, 50, 50));
39     BasicStroke wideStroke = new BasicStroke(2.0f);
40     g2.setColor(Color.black);
41     g2.setStroke(wideStroke);
42     g2.draw(new Ellipse2D.Double(x, y, 50, 50));
43   }
44 }
