19 package org.foo.shape;
20 
21 import java.awt.Graphics2D;
22 import java.awt.Point;
23 
24 /**
25  * This interface defines the <tt>SimpleShape</tt> service. This service is used
26  * to draw shapes. It has two service properties:
27  * <ul>
28  * <li>simple.shape.name - A <tt>String</tt> name for the shape.</li>
29  * <li>simple.shape.icon - An <tt>Icon</tt> for the shape.</li>
30  * </ul>
31  **/
32 public interface SimpleShape {
33 
34   /**
35    * A service property for the name of the shape.
36    **/
37   public static final String NAME_PROPERTY = "simple.shape.name";
38 
39   /**
40    * A service property for the icon of the shape.
41    **/
42   public static final String ICON_PROPERTY = "simple.shape.icon";
43 
44   /**
45    * Draw this shape at the given position.
46    * 
47    * @param g2 The graphics object used for painting.
48    * @param p The position to paint the shape.
49    **/
50   public void draw(Graphics2D g2, Point p);
51 }
