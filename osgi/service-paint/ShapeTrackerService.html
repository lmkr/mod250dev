 19 package org.foo.paint;
 20 
 21 import javax.swing.Icon;
 22 import javax.swing.SwingUtilities;
 23 import org.foo.shape.SimpleShape;
 24 import org.osgi.framework.*;
 25 import org.osgi.util.tracker.ServiceTracker;
 26 
 27 /**
 28  * Extends the <tt>ServiceTracker</tt> to create a tracker for
 29  * <tt>SimpleShape</tt> services. The tracker is responsible for listener for
 30  * the arrival/departure of <tt>SimpleShape</tt> services and informing the
 31  * application about the availability of shapes. This tracker forces all
 32  * notifications to be processed on the Swing event thread to avoid
 33  * synchronization and redraw issues.
 34  **/
 35 public class ShapeTracker extends ServiceTracker {
 36   // Flag indicating an added shape.
 37   private static final int ADDED = 1;
 38   // Flag indicating a modified shape.
 39   private static final int MODIFIED = 2;
 40   // Flag indicating a removed shape.
 41   private static final int REMOVED = 3;
 42   // The bundle context used for tracking.
 43   private BundleContext m_context;
 44   // The application object to notify.
 45   private PaintFrame m_frame;
 46 
 47   /**
 48    * Constructs a tracker that uses the specified bundle context to track
 49    * services and notifies the specified application object about changes.
 50    * 
 51    * @param context The bundle context to be used by the tracker.
 52    * @param frame The application object to notify about service changes.
 53    **/
 54   public ShapeTracker(BundleContext context, PaintFrame frame) {
 55     super(context, SimpleShape.class.getName(), null);
 56     m_context = context;
 57     m_frame = frame;
 58   }
 59 
 60   /**
 61    * Overrides the <tt>ServiceTracker</tt> functionality to inform the
 62    * application object about the added service.
 63    * 
 64    * @param ref The service reference of the added service.
 65    * @return The service object to be used by the tracker.
 66    **/
 67   public Object addingService(ServiceReference ref) {
 68     SimpleShape shape = new DefaultShape(m_context, ref);
 69     processShapeOnEventThread(ADDED, ref, shape);
 70     return shape;
 71   }
 72 
 73   /**
 74    * Overrides the <tt>ServiceTracker</tt> functionality to inform the
 75    * application object about the modified service.
 76    * 
 77    * @param ref The service reference of the modified service.
 78    * @param svc The service object of the modified service.
 79    **/
 80   public void modifiedService(ServiceReference ref, Object svc) {
 81     processShapeOnEventThread(MODIFIED, ref, (SimpleShape) svc);
 82   }
 83 
 84   /**
 85    * Overrides the <tt>ServiceTracker</tt> functionality to inform the
 86    * application object about the removed service.
 87    * 
 88    * @param ref The service reference of the removed service.
 89    * @param svc The service object of the removed service.
 90    **/
 91   public void removedService(ServiceReference ref, Object svc) {
 92     processShapeOnEventThread(REMOVED, ref, (SimpleShape) svc);
 93     ((DefaultShape) svc).dispose();
 94   }
 95 
 96   /**
 97    * Processes a received service notification from the <tt>ServiceTracker</tt>,
 98    * forcing the processing of the notification onto the Swing event thread if
 99    * it is not already on it.
100    * 
101    * @param action The type of action associated with the notification.
102    * @param ref The service reference of the corresponding service.
103    * @param shape The service object of the corresponding service.
104    **/
105   private void processShapeOnEventThread(int action, ServiceReference ref, SimpleShape shape) {
106     if ((m_context.getBundle(0).getState() & (Bundle.STARTING | Bundle.ACTIVE)) == 0) {
107       return;
108     }
109 
110     try {
111       if (SwingUtilities.isEventDispatchThread()) {
112         processShape(action, ref, shape);
113       } else {
114         SwingUtilities.invokeAndWait(new ShapeRunnable(action, ref, shape));
115       }
116     } catch (Exception ex) {
117       ex.printStackTrace();
118     }
119   }
120 
121   /**
122    * Actually performs the processing of the service notification. Invokes the
123    * appropriate callback method on the application object depending on the
124    * action type of the notification.
125    * 
126    * @param action The type of action associated with the notification.
127    * @param ref The service reference of the corresponding service.
128    * @param shape The service object of the corresponding service.
129    **/
130   private void processShape(int action, ServiceReference ref, SimpleShape shape) {
131     String name = (String) ref.getProperty(SimpleShape.NAME_PROPERTY);
132 
133     switch (action) {
134       case MODIFIED:
135         m_frame.removeShape(name);
136         // Purposely let this fall through to the 'add' case to
137         // reload the service.
138 
139       case ADDED:
140         Icon icon = (Icon) ref.getProperty(SimpleShape.ICON_PROPERTY);
141         m_frame.addShape(name, icon, shape);
142         break;
143 
144       case REMOVED:
145         m_frame.removeShape(name);
146         break;
147     }
148   }
149 
150   /**
151    * Simple class used to process service notification handling on the Swing
152    * event thread.
153    **/
154   private class ShapeRunnable implements Runnable {
155     private int m_action;
156     private ServiceReference m_ref;
157     private SimpleShape m_shape;
158 
159     /**
160      * Constructs an object with the specified action, service reference, and
161      * service object for processing on the Swing event thread.
162      * 
163      * @param action The type of action associated with the notification.
164      * @param ref The service reference of the corresponding service.
165      * @param shape The service object of the corresponding service.
166      **/
167     public ShapeRunnable(int action, ServiceReference ref, SimpleShape shape) {
168       m_action = action;
169       m_ref = ref;
170       m_shape = shape;
171     }
172 
173     /**
174      * Calls the <tt>processShape()</tt> method.
175      **/
176     public void run() {
177       processShape(m_action, m_ref, m_shape);
178     }
179   }
180 }
